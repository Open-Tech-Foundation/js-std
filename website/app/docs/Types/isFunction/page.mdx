Checks if the given value is a [Function](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function).

## Syntax

```ts
import { isFunction } from '@opentf/std';

isFunction(val: unknown): val is (...args: unknown[]) => unknown
```

## Examples

```ts
isFunction(function() {}) //=> true

isFunction({}) //=> false

function* generator(i) {
  yield i;
  yield i + 10;
}

isFunction(generator); //=> true
```
