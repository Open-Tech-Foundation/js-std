> Set the property value at object path.

## Syntax

```ts
import { setInObj } from '@open-tech-world/es-utils';

setInObj(obj: ObjType, path: string, value: unknown): ObjType;
```

## Examples

```ts
setInObj(); // It throws an error

setInObj({}, ''); // {}

setInObj({}, 'a'); // { a: undefined }

setInObj({}, 'a', null); // { a: null }

setInObj({}, 'a', 1); // { a: 1 }

setInObj({}, 'a.b', 25); // { a: { b: 25 } }

setInObj({}, 'user.email', 'user@example.com'); // { user: { email: 'user@example.com' } }

const obj = { name: 'x' };
setInObj(obj, 'name', 'xxx');
console.log(obj.name); // 'xxx'

setInObj({}, 'fruits[0]', 'Apple'); // { fruits: ['Apple'] }

setInObj({ fruits: ['Apple'] }, 'fruits[0]', 'Mango'); // { fruits: ['Mango'] }

setInObj({ fruits: ['Apple'] }, 'fruits[1]', 'Mango'); // { fruits: ['Apple', 'Mango'] }

setInObj({ a: [{ b: { c: 3 } }] }, 'a[0].b.c', 4); // { a: [{ b: { c: 4 } }] }
```
