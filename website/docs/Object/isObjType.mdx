> Check whether the given value is an [Object](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Object) type.

## Syntax

```ts
import { isObjType } from '@open-tech-world/es-utils';

isObjType(obj: unknown): boolean;
```

:::caution
isObjType(new functionName()) returns true.
:::

## Examples

```ts
isObjType(); // false

isObjType(undefined); // false

isObjType(null); // false

isObjType(true); // false

isObjType(false); // false

isObjType(0); // false

isObjType(1n); // false

isObjType(''); // false

isObjType([]); // false

isObjType(() => {}); // false

isObjType(Symbol); // false

isObjType(Symbol('Sym')); // false

isObjType(Math); // false

isObjType(Math.min); // false

isObjType(Error); // false

isObjType(new Error()); // false

function fun(x) {
  return x;
}

isObjType(fun); // false

isObjType(fun()); // false

isObjType(Map); // false

isObjType(new Map()); // false

isObjType(Set); // false

isObjType(new Set()); // false

isObjType(/./); // false

isObjType({}); // true

isObjType({ a: 1 }); // true

isObjType(new fun(1)); // true

isObjType(new Object()); // true

isObjType(Object.create(null)); // true

```
