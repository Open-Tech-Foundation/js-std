> Removes all the [falsy](https://developer.mozilla.org/en-US/docs/Glossary/Falsy) values in the given array.

## Syntax

```ts copy
import { arrCompact } from '@opentf/utils';

arrCompact<T>(arr: T[]): Partial<T[]>;
```

## Examples

```ts
arrCompact([undefined]) //=>[]

arrCompact([undefined, null]) //=>[]

arrCompact([undefined, null, 0]) //=>[]

arrCompact([undefined, null, 0, 1]) //=>[1]

arrCompact([undefined, null, 0, 1, -0, 2]) //=>[1, 2]

arrCompact([undefined, null, 0, 1, -0, 2, false, 3]) //=> [1, 2, 3];

arrCompact([undefined, null, 0, 1, -0, 2, false, 3, true, NaN]) //=> [1, 2, 3, true]

arrCompact(['apple', '', 'Mango']) //=>['apple', 'Mango']);
```
