## Syntax

```ts copy
import { arrDiff } from '@opentf/utils';

arrDiff(arr1: unknown[], arr2: unknown[]): unknown[];
```

## Examples

```ts
arrDiff([1], []); // [1]

arrDiff([1], [1, 2]); // []

arrDiff([1, "a"], [1, 2]); // ['a']

arrDiff([0], [0]); // []

arrDiff([-0], [0]); // [-0]

arrDiff([], []); // []

arrDiff(["apple", "mango", "orange"], ["mango", "apple"]); // ['orange']

arrDiff([{ a: 1 }, {a: 3}], [{ a: 1 }, { a: 2 }]); //=> [{a: 3}]
```
