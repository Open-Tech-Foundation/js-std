> Splits the given array values into groups of a given size.

## Syntax

```ts copy
import { chunk } from '@opentf/utils';

chunk(arr: unknown[], size = 1): unknown[];
```

## Examples

```ts
chunk([1, 2, 3]) //=> [[1], [2], [3]]

chunk([1, 2, 3], 1) //=> [[1], [2], [3]]

chunk([1, 2, 3], 2) //=> [[1, 2], [3]]

chunk([1, 2, 3], 3) //=> [[1, 2, 3]]

chunk([1, 2, 3], 4) //=> [[1, 2, 3]]
```
