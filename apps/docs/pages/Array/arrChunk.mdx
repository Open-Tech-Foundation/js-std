> Splits the given array values into groups of a given size.

## Syntax

```ts copy
import { arrChunk } from '@opentf/utils';

arrChunk(arr: unknown[], size = 1): unknown[];
```

## Examples

```ts
arrChunk([1, 2, 3]) //=> [[1], [2], [3]]

arrChunk([1, 2, 3], 1) //=> [[1], [2], [3]]

arrChunk([1, 2, 3], 2) //=> [[1, 2], [3]]

arrChunk([1, 2, 3], 3) //=> [[1, 2, 3]]

arrChunk([1, 2, 3], 4) //=> [[1, 2, 3]]
```
