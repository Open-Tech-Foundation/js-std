## Syntax

```ts copy
import { unescapeHTML } from '@opentf/std';

unescapeHTML(str: string): string;
```

## Returns

The unescaped string.

## Examples

```ts
unescapeHTML('&lt;script&gt;alert(&quot;xss&quot;)&lt;/script&gt;')
//=> '<script>alert("xss")</script>'
```
