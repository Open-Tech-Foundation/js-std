> 🔒 Escapes characters for use in HTML.

## Syntax

```ts copy
import { escapeHTML } from '@opentf/std';

escapeHTML(str: string): string
```

## Examples

```ts
escapeHTML('<script>alert("xss")</script>')
//=> '&lt;script&gt;alert(&quot;xss&quot;)&lt;/script&gt;'
```
