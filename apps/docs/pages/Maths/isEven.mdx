> Checks if the given number is an `Even` number.

## Syntax

```ts
import { isEven } from '@opentf/std';

isEven(n: number): boolean
```

## Examples

```ts
isEven(0) //=> true

isEven(1) //=> false

isEven(2) //=> true

isEven(-1) //=> false

isEven(-2) //=> true
```
