> Checks if the given number is a `Odd` number.

## Syntax

```ts
import { isOdd } from '@opentf/std';

isOdd(n: number): boolean
```

## Examples

```ts
isOdd(0) //=> false

isOdd(1) //=> true

isOdd(2) //=> false

isOdd(-1) //=> true

isOdd(-2) //=> false
```
