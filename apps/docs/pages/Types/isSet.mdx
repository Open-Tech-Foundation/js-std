> Checks if the given value is [Set](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Set) object.

## Syntax

```ts
import { isSet } from '@opentf/utils';

isSet(value: unknown): boolean;
```

## Examples

```ts
isSet(new Set()) //=> true

isSet(new WeakSet()) //=> false

isSet({}) //=> false
```
