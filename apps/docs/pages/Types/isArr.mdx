> Checks if the given value is an [Array](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array).

## Syntax

```ts
import { isArr } from '@opentf/std';

isArr(val: unknown): boolean;
```

## Examples

```ts
isArr([]) //=> true
 
isArr(new Array('a', 'b', 'c')) //=> true

isArr('abc') //=> false
```
