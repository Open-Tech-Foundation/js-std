> Checks if the given value is [null](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Operators/null).

## Syntax

```ts
import { isNull } from '@opentf/utils';

isNull(val: unknown): boolean;
```

## Examples

```ts
isNull(null) //=> true

isNull(undefined) //=> false
```
