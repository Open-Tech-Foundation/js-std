> Checks if the given value is an [ArrayBuffer](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/ArrayBuffer).

## Syntax

```ts
import { isArrBuf } from '@opentf/utils';

isArrBuf(val: unknown): boolean;
```

## Examples

```ts
isArrBuf([]) //=> false

isArrBuf(new ArrayBuffer(8)) //=> true
```
