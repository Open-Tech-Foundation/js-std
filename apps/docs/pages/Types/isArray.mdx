> Checks if the given value is an [Array](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array).

## Syntax

```ts
import { isArray } from '@opentf/std';

isArray(val: unknown): boolean;
```

## Examples

```ts
isArray([]) //=> true
 
isArray(new Array('a', 'b', 'c')) //=> true

isArray('abc') //=> false
```
