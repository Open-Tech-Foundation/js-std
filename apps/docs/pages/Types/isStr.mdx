> Checks if the given value is [String](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String).

## Syntax

```ts
import { isStr } from '@opentf/utils';

isStr(value: unknown): boolean;
```

## Examples

```ts
isStr('abc') //=> true

isStr(String(1)) //=> true

isStr(1) //=> false
```
