> Checks if the given value is [Symbol](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Symbol).

## Syntax

```ts
import { isSym } from '@opentf/std';

isSym(value: unknown): boolean;
```

## Examples

```ts
isSym(Symbol()) //=> true

isSym(Symbol('abc')) //=> true

isSym('abc') //=> false
```
