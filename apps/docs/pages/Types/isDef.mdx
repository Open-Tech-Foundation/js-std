> Checks if the given value is defined.

## Related

- [isUndef](/Types/isUndef)

## Syntax

```ts
import { isDef } from '@opentf/utils';

isDef(value: unknown): boolean;
```

## Examples

```ts
isDef() //=> false

isDef(undefined) //=> false

isDef(null) //=> true

isDef('') //=> true

isDef(0) //=> true

isDef([]) //=> true

isDef({}) //=> true
```
