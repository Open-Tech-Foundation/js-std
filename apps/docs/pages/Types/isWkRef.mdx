> Checks if the given value is [WeakRef](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/WeakRef).

## Syntax

```ts
import { isWkRef } from '@opentf/utils';

isWkRef(value: unknown): boolean;
```

## Examples

```ts
isWkRef({}) //=> false

let user = { name: "John" };
isWkRef(new WeakRef(user)) //=> true
```
