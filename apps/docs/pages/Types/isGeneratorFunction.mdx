> Checks if the given value is a [Generator Function](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/function*).

## Syntax

```ts
import { isGeneratorFunction } from '@opentf/std';

isGeneratorFunction(
  val: unknown
): boolean
```

## Examples

```ts
isGeneratorFunction(function() {}) //=> false

isGeneratorFunction({}) //=> false

function* generator(i) {
  yield i;
  yield i + 10;
}

isGeneratorFunction(generator); //=> true

async function* aGen(i) {
  yield i;
  yield i + 10;
}

isGeneratorFunction(aGen); //=> true

const someObj = {
  *generator() {
    yield "a";
    yield "b";
  },
};

isGeneratorFunction(someObj.generator); //=> true
```
