> Converts the given string into an object property path array.

## Syntax

```ts
import { toPath } from '@opentf/utils';

toPath(str: string): (string | number)[]
```

## Examples

```ts
toPath('a[0].b.c') //=> ['a', 0, 'b', 'c']
```
