> Gets a property value from the object path.

## Syntax

```ts
import { getInObj } from '@opentf/utils';

getInObj(obj: ObjType, path: string): unknown;
```

## Examples

```ts
getInObj(); // It throws an error

getInObj({}, ''); // undefined

getInObj({}, 'a'); // undefined

getInObj({ a: null }, 'a'); // null

getInObj({ a: 1 }, 'a'); // 1

getInObj({ a: { b: 25 } }, 'a.b'); // 25

getInObj({ user: { email: 'user@example.com' } }, 'user.email'); // 'user@example.com'

getInObj({ fruits: ['Apple'] }, 'fruits[0]'); // 'Apple'

getInObj({ fruits: ['Apple', 'Mango'] }, 'fruits'); // [ 'Apple', 'Mango' ]

getInObj({ a: [{ b: { c: 99 } }] }, 'a[0].b.c'); // 99
```
