> It deeply clones the given value.

## Syntax

```ts
import { clone } from '@opentf/utils';

clone(obj: unknown): unknown;
```

<Callout type="info">
The following types are supported:

- Primitive values:
    - Boolean
    - Null
    - Undefined
    - Number
    - BigInt
    - String

- Objects

- Arrays

- Dates
</Callout>

## Examples

```ts
clone(); // undefined

clone(undefined); // undefined

clone(null); // null

clone(true); // true

clone(false); // false

clone('abc'); // 'abc'

clone(1); // 1

clone(1n); // 1n

let input = [];
let output = clone(input);
console.log(output === input); // false

input = [1];
output = clone(input);
console.log(output === input); // false

input = [1, 'abc', false, null, undefined];
output = clone(input);
console.log(output === input); // false

input = {};
output = clone(input);
console.log(output === input); // false

input = { a: undefined, b: null, c: '', d: true, e: 1, f: 5n };
output = clone(input);
console.log(output === input); // false

input = { arr: [1, 2, 3], obj: { a: 'abc', b: 123 } };
output = clone(input);
console.log(output === input); // false
console.log(output.arr === input.arr); // false
console.log(output.obj === input.obj); // false

input = new Date();
output = clone(input);
console.log(output === input); // false

input = new Date('2020-05-01');
output = clone(input);
console.log(output === input); // false
```
