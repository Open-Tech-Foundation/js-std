> The No Operation function.

## Syntax

```ts
import { noop } from '@opentf/utils';

noop(): undefined;
```

## Examples

```ts
noop(); //=> undefined

function test(cb) {
  const fn = cb || noop;
  fn();
}

test(); // It can safely calls the callback fn
```
