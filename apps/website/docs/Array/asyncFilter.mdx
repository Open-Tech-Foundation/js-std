> Array filter with `Async` callback function.

## Syntax

```ts
asyncFilter(arr: unknown[], cb: (value: unknown, index: number) => Promise<boolean>): Promise<unknown[]>
```

## Usage

```ts
import { asyncFilter } from "@opentf/utils";

asyncFilter(arr, async (value, index) => {});
```

## Examples

```ts
const arr = [1, 2, 3, 4, 5];

function isEven(n) {
  return new Promise((resolve) => {
    resolve(n % 2 === 0);
  });
}

const filteredArr = await asyncFilter(arr, async (n) => await isEven(n));

console.log(filteredArr); //=> [2, 4]
```
